\begin{quotation}
  $\lambda (x, y). x \cdot y$ is primitive recursive.
\end{quotation}

\begin{proof}
Take $\lambda (x, y). x \cdot y = P(\lambda x.0, (\lambda (x, y). x + y) (e^{3}_{1}, e^{3}_{2}))$.
\end{proof}
