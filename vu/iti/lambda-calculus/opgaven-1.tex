\item % 10

\begin{description}

\item{(a)}
$\underline{(\lambda y. \textsf{mul} \, 3 \, y) \; 7}
  \quad \rightarrow_{\beta} \quad
  \textsf{mul} \, 3 \, 7
  \quad \rightarrow_{\delta} \quad
  21$

\item{(b)}
$(\underline{(\lambda x \, y. \textsf{mul} \, x \, y) \, 3}) \, 7
  \quad \rightarrow_{\beta} \quad
  \underline{(\lambda y. \textsf{mul} \, 7 \, y) \, 3}
  \quad \rightarrow_{\beta} \quad
  \underline{\textsf{mul} \, 7 \, 3}
  \quad \rightarrow_{\delta} \quad
  21$

\item{(c)}
\begin{align*}
  \underline{(\lambda x. x) \; (\lambda y. \textsf{mul} \, 3 \, y)} \; 7
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda y. \textsf{mul} \, 3 \, y) \; 7} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{\textsf{mul} \, 3 \, 7} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{\textsf{mul} \, 3 \, 7} \\
  & \quad \rightarrow_{\delta} \quad
  21
\end{align*}

\item{(d)}
\begin{align*}
  \underline{(\lambda f \, x. f \, x) \quad (\lambda y. \textsf{plus} \, x \, y)} \quad 3
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda z. \underline{(\lambda y. \textsf{plus} \, x \, y) \, z}) \quad 3} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda z. \textsf{plus} \, x \, z) \quad 3} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{\textsf{plus} \, x \, 3}
\end{align*}

\item{(e)}
\begin{align*}
  \underline{(\lambda f. f \, 3) \; (\lambda x. \textsf{mul} \, x \, 7)}
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda x. \textsf{mul} \, x \, 7) \, 3} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{\textsf{mul} \, 3 \, 7} \\
  & \quad \rightarrow_{\delta} \quad
  21
\end{align*}

\item{(f)}
\begin{align*}
  \underline{(\lambda f \, x. f \, (f \, (x))) \; (\lambda x. \textsf{mul} \, x \, 7)} \; 3
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda x. \underline{(\lambda x. \textsf{mul} \, x \, 7) \, (\underline{(\lambda x. \textsf{mul} \, x \, 7) \, (x)})}) \; 3} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda x. \underline{(\lambda x. \textsf{mul} \, x \, 7) \, (\textsf{mul} \, x \, 7)}) \; 3} \\
  & \quad \rightarrow_{\beta} \quad
  \underline{(\lambda x. \textsf{mul} \, (\textsf{mul} \, x \, 7) \, 7) \; 3} \\
  & \quad \rightarrow_{\beta} \quad
  \textsf{mul} \, (\underline{\textsf{mul} \, 3 \, 7}) \, 7 \\
  & \quad \rightarrow_{\delta} \quad
  \underline{\textsf{mul} \, 21 \, 7} \\
  & \quad \rightarrow_{\delta} \quad
  147
\end{align*}

\end{description}
